SELECT 
    COUNT(*) as total_rows_affected,
    COUNT(CASE WHEN prelim_extension_count = 0 THEN 1 END) as prelim_ext_0,
    COUNT(CASE WHEN prelim_extension_count = 1 THEN 1 END) as prelim_ext_1,
    COUNT(CASE WHEN prelim_extension_count = 2 THEN 1 END) as prelim_ext_2,
    COUNT(CASE WHEN prelim_extension_count NOT IN (0,1,2) THEN 1 END) as prelim_ext_other,
    COUNT(CASE WHEN inves_extension = false THEN 1 END) as inves_false,
    COUNT(CASE WHEN inves_extension = true THEN 1 END) as inves_true,
    COUNT(CASE WHEN inves_extension IS NULL THEN 1 END) as inves_null
FROM public."case"
WHERE received_date < DATE '2025-02-19';
